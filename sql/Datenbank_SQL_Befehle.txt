-- Erstellen der Tabelle 'fragebogen'
CREATE TABLE fragebogen(
	id serial NOT NULL,
	zeitstempel timestamp NOT NULL,
	name varchar(255) NOT NULL,
	lehrerid bigint unsigned NOT NULL,
	fachid bigint unsigned NOT NULL,
	klassename varchar(32) NOT NULL,
	schueleranzahl smallint NOT NULL,
	PRIMARY KEY (id),
	FOREIGN KEY (klassename) REFERENCES klasse (name),
	FOREIGN KEY (fachid) REFERENCES fach (id),
	FOREIGN KEY (lehrerid) REFERENCES lehrer (id)
);
-- Erstellen der Tabelle 'codes'
CREATE TABLE codes(
	codehash varchar(16) NOT NULL,
	fragebogenid bigint unsigned NOT NULL,
	PRIMARY KEY (codehash),
	FOREIGN KEY (fragebogenid) REFERENCES fragebogen (id)
);
-- Erstellen der n..m Tabelle 'nm_frage_fragebogen'
CREATE TABLE nm_frage_fragebogen(
	frageid bigint unsigned NOT NULL,
	bogenid bigint unsigned NOT NULL,
	FOREIGN KEY (frageid) REFERENCES fragen (id),
	FOREIGN KEY (bogenid) REFERENCES fragebogen (id),
	UNIQUE (frageid, bogenid)
);
-- Erstellen der n..m Tabelle 'nm_lehrer_klasse'
CREATE TABLE nm_lehrer_klasse(
	lehrerid bigint unsigned NOT NULL,
	klassename varchar(32) NOT NULL,
	fachid bigint unsigned NOT NULL,
	FOREIGN KEY (lehrerid) REFERENCES lehrer (id),
	FOREIGN KEY (klassename) REFERENCES klasse (name),
	FOREIGN KEY (fachid) REFERENCES fach (id)
);
-- Erstellen der Tabelle 'bewertungen'
CREATE TABLE bewertungen(
	id serial NOT NULL,
	frageid bigint unsigned NOT NULL,
	bogenid bigint unsigned NOT NULL,
	bewertung smallint NOT NULL,
	PRIMARY KEY (id),
	FOREIGN KEY (frageid) REFERENCES fragen (id),
	FOREIGN KEY (bogenid) REFERENCES fragebogen (id)	
);
-- Erstellen der Tabelle 'verbesserungen' 
CREATE TABLE verbesserungen(
	id serial NOT NULL,
	bogenid bigint unsigned NOT NULL,
	vorschlag text NOT NULL,
	PRIMARY KEY (id),
	FOREIGN KEY (bogenid) REFERENCES fragebogen (id)
);
